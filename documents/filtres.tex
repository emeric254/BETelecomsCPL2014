        /* calcul des Bk */
        for( k = 1; k <= ordre; k++ )
        {
            Bk[k] = gamma * gamma + Math.sin( k * Math.PI / ordre ) * Math.sin( k * Math.PI / ordre );
        }
    \end{lstlisting}

\section{calcul de Gk }

            Gk[k] = ( 4 * Ak[k-1] * Ak[k] ) / ( Bk[k-1] * Gk[k-1] );
        }
    \end{lstlisting}
